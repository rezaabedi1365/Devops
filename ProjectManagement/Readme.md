│
├── Traditional (سنتی)

│   ├── Models

│   │   │   ├── Task List
│   │   │   │   └── Milestone
│   │   │   └── Gantt Chart
│   │   ├── CPM (Critical Path Method) → مدیریت مسیر بحرانی و وابستگی وظایف
│   │   └── PERT (Program Evaluation and Review Technique) → زمان‌بندی احتمالاتی و مدیریت عدم قطعیت
│   │
│   ├── Tools
│   │   ├── Microsoft Project → برنامه‌ریزی مرحله‌ای، Gantt Chart، Milestone
│   │   ├── Primavera P6 → پروژه‌های مهندسی و ساخت و ساز بزرگ
│   │   ├── Smartsheet → مدیریت پروژه مشابه Excel، زمان‌بندی و گزارش
│   │   ├── Wrike → برنامه‌ریزی مرحله‌ای و گزارش‌گیری
│   │   └── TeamGantt → Gantt Chart
│
└── Agile (چابک)
    │
    ├── Models
    │   ├── Scrum
    │   │   ├── Roles / نقش‌ها
    │   │   │   ├── Product Owner → مدیریت نیازمندی‌ها و اولویت‌بندی Backlog
    │   │   │   ├── Scrum Master → راهنمای تیم، حذف موانع، تسهیل فرآیند Scrum
    │   │   │   └── Development Team → تیم اجرایی که کارها را انجام می‌دهد
    │   │   │
    │   │   ├── Artifacts / ابزارها
    │   │   │   ├── Product Backlog → لیست تمام نیازمندی‌ها، وظایف و ویژگی‌های محصول
    │   │   │   ├── Sprint Backlog → مجموعه‌ای از وظایف Backlog که در یک Sprint انجام می‌شود
    │   │   │   └── Increment → نتیجه نهایی یا محصول قابل استفاده در پایان Sprint
    │   │   │
    │   │   ├── Events / رویدادها
    │   │   │   ├── Sprint → بازه زمانی کوتاه (مثلاً ۱–۴ هفته) برای انجام مجموعه‌ای از کارها
    │   │   │   ├── Sprint Planning → جلسه برنامه‌ریزی برای انتخاب Backlog Items
    │   │   │   ├── Daily Standup → جلسه روزانه تیم برای بررسی پیشرفت
    │   │   │   ├── Sprint Review → جلسه بازبینی محصول تولید شده
    │   │   │   └── Sprint Retrospective → جلسه بازخورد و بهبود فرآیند
    │   │   │
    │   │   └── Board
    │   │       └── Task Board / Scrum Board → تخته بصری جریان کار (To Do / In Progress / Done)
    │   │           └── نمایش وضعیت Backlog Items
    │   │
    │   ├── Kanban → مدیریت بصری جریان کار، محدودیت WIP
    │   ├── Scrumban → ترکیب Scrum و Kanban
    │   ├── XP (Extreme Programming) → کیفیت کد، Pair Programming، TDD، Continuous Integration
    │   └── Lean → بهینه‌سازی فرآیند، جریان ارزش، حذف اتلاف
    │
    └── Tools
        ├── Jira → مدیریت Backlog، Sprint، Kanban Board، گزارش Agile
        ├── Trello → Board بصری، کارت‌ها و ستون‌های To Do/In Progress/Done
        ├── Azure DevOps → Backlog، Sprint، Board و CI/CD
        ├── Monday.com → مدیریت بصری، Board و Timeline
        ├── ClickUp → ترکیبی Agile/Waterfall، Board، Gantt، Sprint
        ├── Asana → مدیریت Taskها، Board و Timeline
        ├── GitLab
        │       ├── Issue Boards
        │       ├── Epics و Milestones
        │       ├── Task Lists داخل Issue
        │       └── Roadmaps
        └── GitHub
                ├── Project Boards
                ├── Issues
                └── Milestones
